CREATE DATABASE IF NOT EXISTS RESTAURANTE;
USE RESTAURANTE;

CREATE TABLE Cliente (
CPF VARCHAR(18) PRIMARY KEY,
Nome VARCHAR(20) NOT NULL,
Sobrenome VARCHAR(20) NOT NULL
);

CREATE TABLE Pedido (
Idpedido SMALLINT AUTO_INCREMENT PRIMARY KEY,
InicioPedido DATETIME,
FimPedido DATETIME,
TempoPedido DATETIME
);

CREATE TABLE Mesa (
Num SMALLINT AUTO_INCREMENT PRIMARY KEY,
Quant SMALLINT
);

CREATE TABLE PratosBebidas (
Idpratosbebidas SMALLINT AUTO_INCREMENT PRIMARY KEY,
Nome VARCHAR(50),
Preço FLOAT
);

CREATE TABLE Atendente (
Matricula SMALLINT AUTO_INCREMENT PRIMARY KEY,
Nome VARCHAR(20),
Salário FLOAT,
Contato VARCHAR(15)
);

CREATE TABLE AtendenteSenior(
Matricula SMALLINT AUTO_INCREMENT PRIMARY KEY,
Nome VARCHAR(20),
Salário FLOAT,
Contato VARCHAR(15),
CONSTRAINT fk_atendente FOREIGN KEY (Matricula) REFERENCES Atendente(Matricula)
) ENGINE=InnoDB;
